 23	def handlePackets(self, packet: dict, client: Client):
 24		""" Called internally when a packed comes in after a keepalive is sent"""
 25
 26		if packet.get("cmd") == "alert":
 27			if packet.get("usedCommand") is not None:
 28				cmd = packet["usedCommand"].split(" ")[0]
 29				print(cmd)
 30				if not cmd in self.clientCommands:
 31					print("ERROR", packet, "is not a valid command")
 32				else:
 33					func, doThread = self.clientCommands[cmd]
 34					if doThread:
 35						Thread(target=func, args=(self, packet["usedCommand"], )).start()
 36					else:
 37						client.withStatementDeph+=1
 38						try:
 39							func(self, packet["usedCommand"])
 40						finally:
 41							client.withStatementDeph-=1
 42		elif packet.get("cmd") == "keep going":
 43
 44			self.lastX = packet["x"]
 45			self.lastY = packet["y"]
 46			self.lastZ = packet["z"]
 47			self.lastHunger = packet["hunger"]
 48			self.lastHealth = packet["health"]
 49			self.lastYaw = packet["yaw"]
 50			self.lastPitch = packet["pitch"]
 51			self.isFalling = self.lastY < self._prevY
 52			self._prevY=self.lastY
 53
 54			# print(self.lastY, self.isFalling, "falling")
 55			self.lastVelocity = (packet["velx"], packet["vely"], packet["velz"])

 57	def __init__(self, port: int, keepAliveTime=0.1):
 58		"""Port should come from the game's chat. KeepAliveTime is the time between keepalive packets, AKA updates in player info. REQUIRES BARITONE"""
 59
 60		self.client = Client(keepAliveTime, port, appendHandler=self.handlePackets)
 61		with self.client:
 62			self.client.send({"cmd": "is baritone installed"})
 63			self.hasBaritone = self.client.recv(1024)[0].get("isInstalled")
 64
 65		self.clientCommands = {}
 66
 67		self.inventoryManager = InventoryManager(self.client)
