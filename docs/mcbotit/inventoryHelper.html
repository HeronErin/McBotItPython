 17	def __init__(self, client: Client, nbt: dict):
 18		assert nbt.get("container type") == self.requiredType
 19		self.client = client
 20		self.items = nbt["items"]
 21		self.init()
 22	def getId(self, id : int):
 23		""" From a slot id, get an itemstack"""
 24		x = [s for s in self.items if s["id"] == id]
 25		return x[0]
 26	def init(self):
 27		""" Do not call me """
 28		pass
 29	def getSlotIdByRowCol(self, row : int, col : int) -> int:
 30		""" Use math to get the slot id by the row and col"""
 31
 32		assert row < self.maxRows
 33		assert col < self.maxCols
 34		return self.playerInvEnd-(row+1)*self.maxCols+1+col
 35	def getSlot(self, row : int, col : int) -> dict:
 36		""" Got an itemstack from by a row and col"""
 37
 38		return self.getId(self.getSlotIdByRowCol(row, col))
 39	def serialize(self) -> dict:
 40		out = self.__dict__.copy()
 41		del out["client"]
 42		out["container type"] = self.requiredType
 43		return out
 44	def search(self, id:str) -> Generator[dict, None, None]:
 45		""" Search for an item stack in a chest or non player inventory, YEILD the results"""
 46		for row in range(3, self.maxRows):
 47			for col in range(0, self.maxCols):
 48				if (item:=self.getSlot(row, col))["type"] == id:
 49					yield item
 50	def searchPlayerInv(self, id:str) -> dict:
 51		""" Search specificly the players inventory and RETURN the result"""
 52		for row in range(0, 4):
 53			for col in range(0, 9):
 54				slot = self.getSlot( row, col)
 55				if slot["type"] == id:
 56					return slot
